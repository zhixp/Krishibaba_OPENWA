def parse_whatsapp_payload(payload):
    try:
        message = payload["entry"][0]["changes"][0]["value"]["messages"][0]; user_id = message["from"]; msg_type = message["type"]
        if msg_type == "text": return user_id, message["text"]["body"], msg_type, None
        elif msg_type in ["image", "audio"]: return user_id, f"{msg_type}_received", msg_type, message[msg_type]["id"]
        elif msg_type == "location": return user_id, "location_received", msg_type, {"lat": message["location"].get("latitude"), "lon": message["location"].get("longitude")}
    except (KeyError, IndexError): pass
    return None, None, None, None

def get_coords_from_pincode(pincode, country_code="IN"):
    if not GOOGLE_MAPS_API_KEY: return None, None
    try:
        url = f"https://maps.googleapis.com/maps/api/geocode/json?components=postal_code:{pincode}|country:{country_code}&key={GOOGLE_MAPS_API_KEY}"; res = requests.get(url, timeout=5); res.raise_for_status(); data = res.json()
        if data['status'] == 'OK': return data['results'][0]['geometry']['location'].get('lat'), data['results'][0]['geometry']['location'].get('lng')
    except Exception as e: logging.error(f"Failed to get coords for pincode {pincode} from Google: {e}"); return None, None

def get_weather(lat, lon):
    if not OPENWEATHER_API_KEY: return "⚠️ मौसम सेवा के लिए API की अनुपलब्ध है।", None
    try:
        url = f"https://api.openweathermap.org/data/2.5/forecast?lat={lat}&lon={lon}&appid={OPENWEATHER_API_KEY}&units=metric&lang=hi"; res = requests.get(url, timeout=10); res.raise_for_status(); data = res.json(); forecasts = {}
        for entry in data['list']:
            date = entry['dt_txt'].split(' ')[0]
            if date not in forecasts and len(forecasts) < 3: forecasts[date] = {'temp': entry['main']['temp'], 'desc': entry['weather'][0]['description']}
        forecast_msg = "📍 *3 दिन का मौसम पूर्वानुमान:*"
        for date, val in forecasts.items(): dt_obj = datetime.strptime(date, '%Y-%m-%d'); forecast_msg += f"\n🗓 {dt_obj.strftime('%A')}, {date}: {val['desc']} | 🌡 {val['temp']}°C"
        return forecast_msg.strip(), forecasts
    except Exception as e: logging.error(f"Weather fetch failed: {e}"); return "❌ मौसम की जानकारी प्राप्त करने में असमर्थ।", None

def get_mandi_prices_advanced(crop_name, state_name="madhya pradesh", mandi_name=None, max_results=5):
    crop_slug = CROP_DATABASE.get(crop_name);
    if not crop_slug: return f"❌ माफ करना भैया, मुझे '{crop_name}' फसल की जानकारी मेरे बही-खाते में नहीं मिली।"
    state_slug = STATE_MAP.get(state_name.lower())
    if not state_slug: return f"❌ '{state_name}' राज्य के लिए मंडी भाव उपलब्ध नहीं हैं।"
    base_url = f"https://agriplus.in/prices/{crop_slug}/{state_slug}"; logging.info(f"Starting scrape for base URL: {base_url}"); prices = []
    for page_num in range(1, 8):
        url = f"{base_url}/{page_num}"; logging.info(f"Scraping price page: {url}")
        try:
            res = requests.get(url, timeout=15, headers={'User-Agent': 'Mozilla/5.0'}); res.raise_for_status(); soup = BeautifulSoup(res.text, "html.parser")
            price_table = soup.find("table", class_="price-table")
            if not price_table:
                if page_num == 1: logging.warning("Price table not found on first page."); break
            rows = price_table.find("tbody").find_all("tr")
            if not rows: break
            for row in rows:
                cols = row.find_all("td");
                if len(cols) < 9: continue
                market, variety, min_p, max_p, modal_p, date = cols[2].text.strip(), cols[4].text.strip(), cols[5].text.strip(), cols[6].text.strip(), cols[7].text.strip(), cols[8].text.strip()
                if modal_p == 'N/A': continue
                price_string = f"📍 *{market}*\n   🌱 Variety: {variety}\n   🗓️ Date: {date}\n   💰 *Price: ₹{modal_p}/क्विंटल* (Min: ₹{min_p} | Max: ₹{max_p})"
                if mandi_name and mandi_name.lower() in market.lower(): return f"✅ *{crop_name}* का भाव *{market}* में मिल गया:\n\n{price_string}"
                if len(prices) < max_results: prices.append(price_string)
        except requests.exceptions.HTTPError as e:
            if e.response.status_code == 404: logging.warning(f"Page {page_num} not found."); break
            else: logging.error(f"HTTP error on {url}: {e}"); break
        except Exception as e: logging.error(f"Failed during scrape of {url}: {e}"); break
    if mandi_name:
        if prices: return f"🤔 माफ करना भैया, '{mandi_name}' मंडी का सटीक भाव तो नहीं मिला, पर यह कुछ दूसरी मंडियों के भाव हैं:\n\n" + "\n\n".join(prices)
        else: return f"❌ माफ करना, '{state_name}' में '{crop_name}' के लिए '{mandi_name}' मंडी का कोई भाव नहीं मिला।"
    if prices: return f"📊 *{crop_name}* के कुछ ताज़ा मंडी भाव:\n\n" + "\n\n".join(prices)
    return f"❌ '{state_name}' में '{crop_name}' के लिए कोई मंडी भाव नहीं मिला।"

def parse_user_intent(user_msg, conversation_history):
    history_str = json.dumps(conversation_history)
    prompt = f"""You are an NLU engine for Krishi Baba. Analyze the user's latest message in context of the history to determine their intent.
Categorize into: "weather_query", "mandi_query", "kvk_query", "general_advice_query", or "greeting_or_smalltalk".
History: {history_str}
User Message: "{user_msg}"
Respond with ONLY a JSON object with a single key "intent"."""
    try:
        result = model.generate_content(prompt); clean_text = result.text.strip().replace("```json", "").replace("```", ""); data = json.loads(clean_text); return data
    except Exception as e: logging.error(f"Gemini NLU parsing error: {e}"); return {"intent": "general_advice_query"}

def extract_mandi_details(user_msg, conversation_history):
    crop_list_for_prompt = ", ".join(list(CROP_DATABASE.keys())[:50])
    history_str = json.dumps(conversation_history[-4:])
    prompt = f"""You are a highly accurate data extraction bot. Your ONLY job is to identify a single crop and a single market (mandi) name from the user's latest message, using the conversation history for context if needed.
Official Crop List: {crop_list_for_prompt}
Conversation History: {history_str}
User's Latest Message: "{user_msg}"
Instructions: Analyze the "User's Latest Message". If the crop is mentioned, extract it. If not, look at the "Conversation History". Match the found crop to the "Official Crop List". Extract the market/mandi name ONLY if it's in the latest message. If no crop is found, return null.
Respond with ONLY a JSON object.
Example: "bhav kya hai raisen mandi ka?" -> {{"crop_name": "Soyabean", "mandi": "raisen"}}"""
    try:
        result = model.generate_content(prompt); clean_text = result.text.strip().replace("```json", "").replace("```", ""); data = json.loads(clean_text); return data
    except Exception as e: logging.error(f"Gemini mandi detail extraction error: {e}"); return {}

    else:
        state["history"].append({"role": "user", "parts": [msg]})
        handle_final_state(state, msg, user_id)
    save_user_state(user_id, state)
    return jsonify({"status": "ok"}), 200

def handle_final_state(state, msg, user_id):
    parsed_intent = parse_user_intent(msg, state.get("history", [])); intent = parsed_intent.get("intent")
    bot_response_text = ""
    if intent == "weather_query": handle_weather_query(state, user_id)
    elif intent == "mandi_query": bot_response_text = handle_mandi_query(state, msg, user_id)
    elif intent == "kvk_query": bot_response_text = handle_kvk_query(state, user_id)
    elif intent == "greeting_or_smalltalk": bot_response_text = "राम राम भैया! और बताओ, क्या सेवा करूँ आपकी?"
    else: bot_response_text = handle_general_advice(state, msg, user_id)
    if bot_response_text:
        send_whatsapp_message(user_id, bot_response_text)
        state["history"].append({"role": "model", "parts": [bot_response_text]})

def handle_survey(state, msg, user_id):
    current_step = state.get('step', ''); bot_response_text = ""
    if not current_step.startswith('survey_'):
        state['step'] = 'survey_q1'
        intro = f"नमस्ते {state.get('name', 'किसान भाई')}! कृषि बाबा के काम को और बेहतर बनाने के लिए, क्या आप कुछ छोटे सवालों के जवाब देंगे?"
        send_whatsapp_message(user_id, intro); bot_response_text = SURVEY_QUESTIONS["q1"]
    else:
        question_key = current_step.replace('survey_', ''); save_survey_response(user_id, question_key, msg)
        next_q_num = int(question_key[1:]) + 1; next_q_key = f'q{next_q_num}'
        if next_q_key in SURVEY_QUESTIONS:
            state['step'] = next_q_key; bot_response_text = SURVEY_QUESTIONS[next_q_key]
        else:
            bot_response_text = "बहुत-बहुत धन्यवाद भैया! आपकी सलाह मेरे लिए अनमोल है। 🙏"; state['step'] = 'final'
    send_whatsapp_message(user_id, bot_response_text)
    state["history"].append({"role": "user", "parts": [msg]}); state["history"].append({"role": "model", "parts": [bot_response_text]})

def handle_weather_query(state, user_id):
    lat, lon = state.get("lat"), state.get("lon")
    if not lat and "pincode" in state:
        send_whatsapp_message(user_id, f"आपके पिनकोड {state['pincode']} के लिए मौसम खोजा जा रहा है..."); lat, lon = get_coords_from_pincode(state['pincode'])
    if lat and lon:
        data_forecast, forecast_dict = get_weather(lat, lon); send_whatsapp_message(user_id, data_forecast)
        if forecast_dict:
            commentary_prompt = f"""तुम कृषि बाबा हो, एक 500 साल के ज्ञानी और मज़ाकिया किसान। यह 3 दिन का मौसम डेटा है: {forecast_dict}। इस डेटा के आधार पर, किसान को 1-2 लाइन में देसी अंदाज़ में एक मज़ेदार और उपयोगी सलाह दो।"""
            try: result = model.generate_content(commentary_prompt); send_whatsapp_message(user_id, result.text.strip())
            except Exception as e: logging.error(f"Gemini weather commentary error: {e}")
    else: send_whatsapp_message(user_id, "🌦️ मौसम की जानकारी के लिए, कृपया पहले अपनी लोकेशन या पिनकोड भेजें।")

def handle_mandi_query(state, msg, user_id):
    params = extract_mandi_details(msg, state.get("history", [])); crop_name = params.get("crop_name") or state.get("crop")
    if not crop_name: return "🤔 माफ करना भैया, मैं समझ नहीं पाया आप कौन सी फसल की बात कर रहे हो।"
    mandi = params.get("mandi"); user_state_name = "madhya pradesh"

    return get_mandi_prices_advanced(crop_name, user_state_name, mandi)

def handle_general_advice(state, msg, user_id):

def send_whatsapp_message(recipient_id, message_text):
    if not message_text: return

def initiate_survey_process():
    logging.info("Starting survey job...")
    all_states = get_all_states()
    users_to_survey = []

    for user_id, state in all_states.items():
        if state.get("step") not in ["final", "done"]: continue
        last_survey_time_str = state.get("last_survey_sent")
        if last_survey_time_str:
            last_survey_time = datetime.fromisoformat(last_survey_time_str)
            # Change to timedelta(days=6) for weekly schedule
            if datetime.now() - last_survey_time < timedelta(days=6):
                continue
        users_to_survey.append(user_id)

    if not users_to_survey:
        logging.info("No users eligible for a survey today."); return

    logging.info(f"Found {len(users_to_survey)} users to survey. Sending now...")
    first_question_key = "q1"
    first_question_text = SURVEY_QUESTIONS[first_question_key]

    for user_id in users_to_survey:
        try:
            user_state = all_states[user_id]
            intro_message = f"नमस्ते {user_state.get('name', 'किसान भाई')}! कृषि बाबा के काम को और बेहतर बनाने के लिए, क्या आप कुछ छोटे सवालों के जवाब देंगे?"
            send_whatsapp_message(user_id, intro_message)
            send_whatsapp_message(user_id, first_question_text)

            user_state["step"] = f"survey_{first_question_key}"
            user_state["last_survey_sent"] = datetime.now().isoformat()
            save_user_state(user_id, user_state)
            logging.info(f"Survey initiated for {user_id}")
        except Exception as e:
            logging.error(f"Failed to send survey to {user_id}: {e}")
    logging.info("Survey job finished.")

if __name__ == '__main__':
    initiate_survey_process()

import requests
from bs4 import BeautifulSoup
import json
import logging

# Setup logging to see the progress
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')

def generate_crop_database():
    """
    Scrapes all pages of agriplus.in/price/all to build a JSON database
    of all available crops with their names and URL slugs.
    This version uses the precise HTML structure from your screenshot.
    """
    base_url = "https://agriplus.in/price/all"
    crop_database = {}

    # Loop through pages 1 to 7 to be safe and find all crops
    for page_num in range(1, 8):
        # We don't need to append /1 for the first page
        url = base_url if page_num == 1 else f"{base_url}/{page_num}"
        logging.info(f"Scraping crop list from: {url}")

        try:
            res = requests.get(url, timeout=15, headers={'User-Agent': 'Mozilla/5.0'})
            res.raise_for_status()
            soup = BeautifulSoup(res.text, 'html.parser')

            # --- PERFECTED LOGIC BASED ON YOUR SCREENSHOT ---
            # We select all <a> tags with the exact class 'card-wrapper'
            crop_links_on_page = soup.select('a.card-wrapper')

            if not crop_links_on_page:
                logging.info(f"No more crop links found on page {page_num}. Finalizing.")
                break

            for link in crop_links_on_page:
                href = link.get('href', '')
                # The slug is the last part of the URL, e.g., 'alasande-gram'
                slug = href.strip('/').split('/')[-1]
                # The name is the clean text content of the link
                name = " ".join(link.text.strip().split())

                if slug and name:
                    crop_database[name] = slug

        except requests.exceptions.RequestException as e:
            logging.error(f"Could not fetch page {page_num}. Error: {e}")
            continue

    if not crop_database:
        logging.error("CRITICAL: Failed to scrape any crops, even with the new logic. The website may be blocking the requests.")
    else:
        # Save the scraped data to the JSON file
        with open('agriplus_crops.json', 'w', encoding='utf-8') as f:
            json.dump(crop_database, f, indent=4, ensure_ascii=False)
        logging.info(f"Successfully generated agriplus_crops.json with {len(crop_database)} crops.")

if __name__ == '__main__':
    generate_crop_database()

import json
import os
import logging

STATE_FILE = 'user_states.json'

def get_user_state(user_id):
    if not os.path.exists(STATE_FILE): return {"step": "start", "history": []}
    try:
        with open(STATE_FILE, 'r', encoding='utf-8') as f:
            states = json.load(f)
            user_state = states.get(user_id, {"step": "start", "history": []})
            if "history" not in user_state: user_state["history"] = []
            return user_state
    except (json.JSONDecodeError, FileNotFoundError): return {"step": "start", "history": []}

def save_user_state(user_id, state):
    try:
        if "history" in state: state["history"] = state["history"][-10:]
        all_states = {}
        if os.path.exists(STATE_FILE):
            with open(STATE_FILE, 'r', encoding='utf-8') as f:
                try: all_states = json.load(f)
                except json.JSONDecodeError: all_states = {}
        all_states[user_id] = state
        with open(STATE_FILE, 'w', encoding='utf-8') as f: json.dump(all_states, f, indent=2, ensure_ascii=False)
    except Exception as e: logging.error(f"Failed to save state for user {user_id}: {e}")

def get_all_states():
    """Retrieves all user states from the JSON file."""
    if not os.path.exists(STATE_FILE): return {}
    try:
        with open(STATE_FILE, 'r', encoding='utf-8') as f: return json.load(f)
    except (json.JSONDecodeError, FileNotFoundError): return {}

# generate crop database

import requests
from bs4 import BeautifulSoup
import json
import logging

# Setup logging to see the progress
logging.basicConfig(level=logging.INFO, format='%(asctime)s - %(levelname)s - %(message)s')

def generate_crop_database():
    """
    Scrapes all pages of agriplus.in/price/all to build a JSON database
    of all available crops with their names and URL slugs.
    This version uses the precise HTML structure from your screenshot.
    """
    base_url = "https://agriplus.in/price/all"
    crop_database = {}

    # Loop through pages 1 to 7 to be safe and find all crops
    for page_num in range(1, 8):
        # We don't need to append /1 for the first page
        url = base_url if page_num == 1 else f"{base_url}/{page_num}"
        logging.info(f"Scraping crop list from: {url}")

        try:
            res = requests.get(url, timeout=15, headers={'User-Agent': 'Mozilla/5.0'})
            res.raise_for_status()
            soup = BeautifulSoup(res.text, 'html.parser')

            # --- PERFECTED LOGIC BASED ON YOUR SCREENSHOT ---
            # We select all <a> tags with the exact class 'card-wrapper'
            crop_links_on_page = soup.select('a.card-wrapper')

            if not crop_links_on_page:
                logging.info(f"No more crop links found on page {page_num}. Finalizing.")
                break

            for link in crop_links_on_page:
                href = link.get('href', '')
                # The slug is the last part of the URL, e.g., 'alasande-gram'
                slug = href.strip('/').split('/')[-1]
                # The name is the clean text content of the link
                name = " ".join(link.text.strip().split())

                if slug and name:
                    crop_database[name] = slug

        except requests.exceptions.RequestException as e:
            logging.error(f"Could not fetch page {page_num}. Error: {e}")
            continue

    if not crop_database:
        logging.error("CRITICAL: Failed to scrape any crops, even with the new logic. The website may be blocking the requests.")
    else:
        # Save the scraped data to the JSON file
        with open('agriplus_crops.json', 'w', encoding='utf-8') as f:
            json.dump(crop_database, f, indent=4, ensure_ascii=False)
        logging.info(f"Successfully generated agriplus_crops.json with {len(crop_database)} crops.")

if __name__ == '__main__':
    generate_crop_database()
